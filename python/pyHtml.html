def rep( xs ):
    xs = xs.replace( '&', '&amp;' )
    #xs = xs.replace( '\t', '&#9' )
    xs = xs.replace( '\t', '&ensp;&ensp;&ensp;&ensp;' )
    xs = xs.replace( ' ', '&emsp;' )
    xs = xs.replace( '<', '&lt;' )
    xs = xs.replace( '>', '&gt;' )
    xs = xs.replace( '¥', '&yen;' )
    xs = xs.replace( '←', '&larr;' )
    xs = xs.replace( '↑', '&uarr;' )
    xs = xs.replace( '→', '&rarr;' )
    xs = xs.replace( '↓', '&darr;' )
    xs = xs.replace( 'ˆ', '&circ;' )
    xs = xs.replace( '˜', '&tilde;' )
    xs = xs.replace( '—', '&mdash;' )
    xs = xs.replace( '"', '&quot;' )
    xs = xs.replace( '\n', '<br>\n' )
    return xs
