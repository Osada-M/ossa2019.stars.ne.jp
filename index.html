import inspect

def check(func):
    def f(*args, **kwargs):
        sig = inspect.signature(func)
        for (name, param) in sig.parameters.items():
            print(f'args: {name}\ntype: {repr(param.annotation)}')
        print(f'result: {func(*args, **kwargs)}')
        return func
    return f

# func(~~)
